Compute Fibonacci

This code computes the nth Fibonacci number. It starts with 0 and 1, then repeatedly adds the two previous numbers. 
// Compute the nth Fibonacci number
#include 
using namespace std;

int main() {
    int n;
    cin >> n;

    if(n == 0) {
        cout << 0;
        return 0;
    }

    long long first = 0, second = 1;
    for (int i = 2; i <= n; i++) {
        long long next = first + second;
        first = second;
        second = next;
    }

    cout << second;
    return 0;
}

Repetitions

This code checks a sentence and prints out any letters that appear more than once.

#include 
using namespace std;

    char text[1000];
    cin.getline(text, 1000);
    int len = strlen(text);

                alreadyChecked = true;
                break;

}

Permutations

This code prints all the ways to rearrange the letters of a word using recursion. 
// Print all permutations of a word
#include 
#include 
using namespace std;

void swap(char &a, char &b) {
    char temp = a;
    a = b;
    b = temp;
}

void permute(char s[], int l, int r) {
    if(l == r)
        cout << s << "\n";
    else {
        for(int i = l; i <= r; i++){
            swap(s[l], s[i]);
            permute(s, l+1, r);
            swap(s[l], s[i]); // backtrack
        }
    }
}

int main() {
    char s[1000];
    cin >> s;
    int n = strlen(s);
    permute(s, 0, n-1);
    return 0;
}

Two Knights

This code counts the number of ways to place two knights on an n×n chessboard so that they do not attack each other. 
// Count non-attacking placements for two knights
#include 
#include 
using namespace std;

bool attacks(int x1, int y1, int x2, int y2) {
    int dx = abs(x1 - x2), dy = abs(y1 - y2);
    return (dx == 1 && dy == 2) || (dx == 2 && dy == 1);
}

int main() {
    int n;
    cin >> n;
    long long count = 0;
    for(int x1 = 0; x1 < n; x1++){
        for(int y1 = 0; y1 < n; y1++){
            for(int x2 = 0; x2 < n; x2++){
                for(int y2 = 0; y2 < n; y2++){
                    if(x1 == x2 && y1 == y2) continue;
                    if(!attacks(x1, y1, x2, y2))
                        count++;
                }
            }
        }
    }
    cout << count / 2;
    return 0;
}

Fractional Knapsack

This code tries different fractions (using a fixed step) of each item to pack a bag with limited capacity for maximum value. 
// Solve the Fractional Knapsack problem by trying fractions of each item
#include 
using namespace std;

double bestValue = 0;

void solveFractional(int index, double currentWeight, double currentValue, double capacity, int n, double weights[], double values[], double step) {
    if(index == n){
        if(currentWeight <= capacity && currentValue > bestValue)
            bestValue = currentValue;
        return;
    }
    for(double frac = 0; frac <= 1.0; frac += step){
        double newWeight = currentWeight + frac * weights[index];
        double newValue = currentValue + frac * values[index];
        if(newWeight <= capacity)
            solveFractional(index + 1, newWeight, newValue, capacity, n, weights, values, step);
    }
}

int main(){
    int n;
    double capacity;
    cin >> n >> capacity;
    double weights[1000], values[1000];
    for(int i = 0; i < n; i++){
        cin >> weights[i] >> values[i];
    }
    double step = 0.1;
    solveFractional(0, 0.0, 0.0, capacity, n, weights, values, step);
    cout << bestValue;
    return 0;
}

Counting Rooms

This code counts the number of connected areas (rooms) in a grid where '.' represents open space. 
// Count the number of rooms in a grid
#include 
using namespace std;

int n, m;
char grid[101][101];

void dfs(int i, int j) {
    if(i < 0 || i >= n || j < 0 || j >= m)
        return;
    if(grid[i][j] != '.')
        return;
    grid[i][j] = '#';
    dfs(i+1, j);
    dfs(i-1, j);
    dfs(i, j+1);
    dfs(i, j-1);
}

int main(){
    cin >> n >> m;
    for(int i = 0; i < n; i++){
        cin >> grid[i];
    }
    int rooms = 0;
    for(int i = 0; i < n; i++){
        for(int j = 0; j < m; j++){
            if(grid[i][j] == '.'){
                rooms++;
                dfs(i, j);
            }
        }
    }
    cout << rooms;
    return 0;
}

Josephus Problem

This code simulates removing every kth person from a circle until only one person remains. 
// Solve the Josephus problem by simulating removals in a circle
#include 
using namespace std;

int main(){
    int n, k;
    cin >> n >> k;
    int circle[1000];
    for(int i = 0; i < n; i++){
        circle[i] = i + 1;
    }
    int size = n, index = 0;
    while(size > 1){
        index = (index + k - 1) % size;
        for(int i = index; i < size - 1; i++){
            circle[i] = circle[i+1];
        }
        size--;
    }
    cout << circle[0];
    return 0;
}

Tasks and Deadlines

This code tries every order of tasks (each with a deadline and profit) to find the order that gives the maximum profit. 
// Find the best order of tasks to maximize profit
#include 
using namespace std;

int n;
int deadlines[10], profits[10];
int bestProfit = 0;

void swap(int &a, int &b) {
    int temp = a;
    a = b;
    b = temp;
}

void permute(int pos) {
    if(pos == n) {
        int time = 0, profit = 0;
        for(int i = 0; i < n; i++){
            time++;
            if(time <= deadlines[i])
                profit += profits[i];
        }
        if(profit > bestProfit)
            bestProfit = profit;
        return;
    }
    for(int i = pos; i < n; i++){
        swap(deadlines[pos], deadlines[i]);
        swap(profits[pos], profits[i]);
        permute(pos+1);
        swap(deadlines[pos], deadlines[i]);
        swap(profits[pos], profits[i]);
    }
}

int main(){
    cin >> n;
    for(int i = 0; i < n; i++){
        cin >> deadlines[i] >> profits[i];
    }
    permute(0);
    cout << bestProfit;
    return 0;
}

This code finds the time when the most customers were in the restaurant by checking each time unit.

#include 
using namespace std;

int main(){

    int start[1000], finish[1000];
    int minTime = 1000000000, maxTime = 0;
    for(int i = 0; i < n; i++){

        if(start[i] < minTime)
            minTime = start[i];
        if(finish[i] > maxTime)
            maxTime = finish[i];

    int maxCustomers = 0;

}

Increasing Subsequence

This code finds the longest subsequence of numbers that keeps increasing by trying every possibility. 
// Find the longest increasing subsequence using recursion
#include 
using namespace std;

int best = 0;

void dfs(int idx, int last, int count, int n, int arr[]){
    if(idx == n){
        if(count > best)
            best = count;
        return;
    }
    // Option: do not include current element
    dfs(idx + 1, last, count, n, arr);
    // Option: include it if it is larger than last
    if(arr[idx] > last)
        dfs(idx + 1, arr[idx], count + 1, n, arr);
}

int main(){
    int n;
    cin >> n;
    int arr[1000];
    for(int i = 0; i < n; i++){
        cin >> arr[i];
    }
    dfs(0, -1000000000, 0, n, arr);
    cout << best;
    return 0;
}

Minimizing Coins

This code finds the fewest coins needed to make up a target amount by trying all coin combinations. 
// Find the minimum number of coins needed
#include 
#include 
using namespace std;

int best = INT_MAX;

void solve(int index, int count, int remaining, int n, int coins[]){
    if(remaining == 0){
        if(count < best)
            best = count;
        return;
    }
    if(index >= n)
        return;
    for(int i = 0; i * coins[index] <= remaining; i++){
        solve(index + 1, count + i, remaining - i * coins[index], n, coins);
    }
}

int main(){
    int n, amount;
    cin >> n >> amount;
    int coins[1000];
    for(int i = 0; i < n; i++){
        cin >> coins[i];
    }
    solve(0, 0, amount, n, coins);
    cout << (best == INT_MAX ? -1 : best);
    return 0;
}

Jump Game II

This code finds the minimum number of jumps needed to reach the end of an array, where each number tells you the maximum jump length. 
// Find the minimum number of jumps to reach the end of the array
#include 
#include 
using namespace std;

int minJumps = INT_MAX;

void jump(int index, int count, int n, int arr[]){
    if(index >= n - 1){
        if(count < minJumps)
            minJumps = count;
        return;
    }
    for(int j = 1; j <= arr[index]; j++){
        jump(index + j, count + 1, n, arr);
    }
}

int main(){

    cin >> n;
    int arr[1000];
    for(int i = 0; i < n; i++){
        cin >> arr[i];
    }
    jump(0, 0, n, arr);
    cout << (minJumps == INT_MAX ? -1 : minJumps);
    return 0;
}

Planets Queries

This code checks if two nodes (planets) are connected by using an adjacency matrix and depth-first search. 
// Answer connectivity queries using an adjacency matrix and DFS
#include 
using namespace std;

int n;
int adj[101][101];
bool visited[101];

void dfs(int node){
    visited[node] = true;
    for (int i = 0; i < n; i++){
        if(adj[node][i] == 1 && !visited[i])
            dfs(i);
    }
}

int main(){
    int m;
    cin >> n >> m;
    for (int i = 0; i < n; i++){
        for (int j = 0; j < n; j++){
            adj[i][j] = 0;
        }
    }
    for (int i = 0; i < m; i++){
        int u, v;
        cin >> u >> v;
        adj[u][v] = 1;
        adj[v][u] = 1;
    }
    int q;
    cin >> q;
    while(q--){
        int u, v;
        cin >> u >> v;
        for (int i = 0; i < n; i++)
            visited[i] = false;
        dfs(u);
        cout << (visited[v] ? "Yes" : "No") << "\n";
    }
    return 0;
}

Removing Digits

This code finds the minimum number of steps to reduce a number to 0 by subtracting one of its digits at each step. 
// Find the minimum steps to reduce a number to 0 by removing digits
#include 
#include 
using namespace std;

int solve(int n){
    if(n == 0)
        return 0;
    int best = INT_MAX;
    int temp = n;
    while(temp > 0){
        int d = temp % 10;
        temp /= 10;
        if(d > 0){
            int steps = 1 + solve(n - d);
            if(steps < best)
                best = steps;
        }
    }
    return best;
}

int main(){
    int n;
    cin >> n;
    cout << solve(n);
    return 0;
}

Knight's Tour

This code uses backtracking to try to move a knight on an 8×8 chessboard so that it visits every square exactly once. 
// Try to find a full knight's tour on an 8x8 chessboard
#include 
using namespace std;

int board[8][8];
int dx[8] = {2, 1, -1, -2, -2, -1, 1, 2};
int dy[8] = {1, 2, 2, 1, -1, -2, -2, -1};

bool isValid(int x, int y){
    return (x >= 0 && x < 8 && y >= 0 && y < 8 && board[x][y] == -1);
}

bool solveTour(int x, int y, int move){
    if(move == 64)
        return true;
    for(int i = 0; i < 8; i++){
        int nextX = x + dx[i];
        int nextY = y + dy[i];
        if(isValid(nextX, nextY)){
            board[nextX][nextY] = move;
            if(solveTour(nextX, nextY, move + 1))
                return true;
            board[nextX][nextY] = -1;
        }
    }
    return false;
}

int main(){
    for(int i = 0; i < 8; i++)
        for(int j = 0; j < 8; j++)
            board[i][j] = -1;
    board[0][0] = 0;
    if(solveTour(0, 0, 1)){
        for(int i = 0; i < 8; i++){
            for(int j = 0; j < 8; j++){
                cout << board[i][j] << "\t";
            }
            cout << "\n";
        }
    } else {
        cout << "No solution";
    }
    return 0;
}

Labyrinth

This code searches a maze (grid) for an exit. The maze uses '#' for walls, 'S' for start, and 'E' for exit. 
// Find a path out of a maze using DFS
#include 
using namespace std;

int n, m;
char maze[101][101];
bool visited[101][101];

bool dfs(int x, int y){
    if(x < 0 || x >= n || y < 0 || y >= m)
        return false;
    if(maze[x][y] == '#' || visited[x][y])
        return false;
    if(maze[x][y] == 'E')
        return true;
    visited[x][y] = true;
    if(dfs(x+1, y) || dfs(x-1, y) || dfs(x, y+1) || dfs(x, y-1))
        return true;
    return false;
}

int main(){
    cin >> n >> m;
    for(int i = 0; i < n; i++){
        cin >> maze[i];
    }
    int startX, startY;
    for(int i = 0; i < n; i++){
        for(int j = 0; j < m; j++){
            if(maze[i][j] == 'S'){
                startX = i;
                startY = j;
            }
        }
    }
    cout << (dfs(startX, startY) ? "Path found" : "No path");
    return 0;
}
